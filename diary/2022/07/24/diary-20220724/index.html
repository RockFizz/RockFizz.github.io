安装pthread库

Ubuntu默认不带pthread库，需要安装 

sudo apt-get install glibc-doc  
sudo apt-get install manpages-posix-dev  

安装好后用man -k pthread_create查询就可以找到了

编写源代码 

//hellcpp.cpp:  

#include <iostream>
#include "pthread.h"

using namespace std;

#define NUM_THREADS 5

//线程运行的函数
void* say_hello(void* args)
{
   cout << "Hello rundo" << endl;
   return 0;
}

int main()
{
  cout << "hello world c++ " << endl;  
  //定义线程的id变量，多个变量使用数组
  pthread_t tids[NUM_THREADS];
  for(int i=0;i<NUM_THREADS;i++)
  {
    //参数依次是 创建线程的id、线程参数、调用的函数、传入的函数参数
    int ret = pthread_create(&tids[i],NULL,say_hello,NULL);
    if(ret != 0)
    {
      cout << "pthread_create error: error_code=" << ret << endl;
    }
  }
  pthread_exit(NULL);
  return 0;
}

// 编译源代码 
g++ ./hellcpp.cpp -pthread -o test.o
// 运行  
./test.o

使用 pthread_create() 函数创建了 5 个线程，并接收传入的参数。  
每个线程打印一个 "Hello Runoob!" 消息，并输出接收的参数  
然后调用 pthread_exit() 终止线程。

向线程传递参数

以下实例演示了如何通过结构传递多个参数。您可以在线程回调中传递任意的数据类型，因为它指向 void，

// hellcpp2.cpp
#include <iostream>
#include <cstdlib>
#include "pthread.h"

using namespace std;

#define NUM_THREADS 5

struct thread_data{
  int thread_id;
  char* message;
};

void* PrintHello(void* threadarg)
{
   struct thread_data* my_data;
   my_data = (struct thread_data*) threadarg;

   cout << "Thread ID: " << my_data->thread_id ;
   cout << "Message: " << my_data->message << endl;
   pthread_exit(NULL);
}

int main()
{

  pthread_t threads[NUM_THREADS];
  struct thread_data td[NUM_THREADS];
  int rc;
  int i;
  for(int i=0;i<NUM_THREADS;i++)
  {
    cout << "main(): creating thread: " << i << endl;
    td[i].thread_id=i;
    td[i].message = (char*)("this is message");

    rc = pthread_create(&threads[i],NULL,PrintHello,(void*)&td[i]);
    if(rc)
    {
      cout << "error: unable to create thread " << rc << endl;
      exit(-1);
    }
  }
  pthread_exit(NULL);
  return 0;
}

连接和分离线程

可以使用以下两个函数来连接或分离线程 

pthread_join (threadid, status)   
pthread_detach (threadid)

pthread_join() 子程序阻碍调用程序，直到指定的 threadid 线程终止为止。当创建一个线程时，它的某个属性会定义它是否是可连接的（joinable）或可分离的（detached）。只有创建时定义为可连接的线程才可以被连接。如果线程创建时被定义为可分离的，则它永远也不能被连接。

以下示例演示如何使用pthread_join来等待线程的完成 

#include <iostream>
#include <cstdlib>
#include "pthread.h"
#include <unistd.h>


using namespace std;

#define NUM_THREADS 5

void* wait(void* t)
{
   int i;
   long tid;
   tid=(long)t;
   sleep(2);
   cout << "Sleeping in threade" << endl;
   cout << "Thread with id: " << tid << "...exiting " << endl;
   pthread_exit(NULL);
}

int main()
{
  int rc;
  int i;
  pthread_t threads[NUM_THREADS];
  pthread_attr_t attr;
  void* status;


  pthread_attr_init(&attr);
  pthread_attr_setdetachstate(&attr,PTHREAD_CREATE_JOINABLE);

  for(i=0;i<NUM_THREADS;i++)
  {
    cout << "main() : creating thread " << i << endl;
    rc = pthread_create(&threads[i],NULL,wait,(void*)&i);
    if(rc){
      cout<< "Error: unable to create thread "<< rc << endl;
      exit(-1);
    }
  }

  pthread_attr_destroy(&attr);
  for(i=0;i<NUM_THREADS;i++){
    rc=pthread_join(threads[i],&status);
    if(rc)
    {
      cout << "error: unable to join , " << rc << endl;
      exit(-1);
    }
    cout << "Main: completed thread id: " << i;
    cout << "  exiting with status : " << status << endl;
  }

  cout << "Main: program exiting " << endl;
  pthread_exit(NULL);

}

std::thread

C++11添加了新的标准库std::thread 

std::thread在<thread>头文件中声明，  
因此使用std::thread需要包含<thread>头文件  
编译器使用C+11的编译参数是 -std=c++11  
g++ -std=c++11 test.cpp  

std::thread默认构造函数，创建一个空的std::thread的执行对象  

#include <thread>  
std::thread thread_object(callable)
一个可调用对象可以是以下三种的任何一个  
1. 函数指针  
2. 函数对象  
3. lambda表达式  
定义callable后，将其传递给std::thread构造函数thread_object

下例演示多线程cpp，使用是三个不同的可调用对象 

#include <iostream>
#include <thread>
using namespace std;

void foo(int Z){
  for(int i=0;i<Z;i++)
  {
    cout << "thread use fun zhizhen as param\n";
  }
}

class thread_obj
{
public:
  void operator()(int x){
    for(int i=0;i<x;i++){
      cout << "thread use fun obj as param \n";
    }
  }
};

int main(){
  cout << "thread 1 2 3 run independency" << endl;
  thread th1(foo,3);
  thread th2(thread_obj(),3);
  auto f = [](int x){
    for(int i=0;i<x;i++){
      cout << "thread use lambda as param \n ";
    }
  };
  thread th3(f,3);

  th1.join();
  th2.join();
  th3.join();
  return 0;
}

下例是c++11中标准线程库的用法(编译器使用C+11的编译参数是 -std=c++11)

//mstl.cpp :
#include <iostream>
#include <thread>

std::thread::id main_thread_id = std::this_thread::get_id();

void hello()  
{
    std::cout << "Hello Concurrent World\n";
    if (main_thread_id == std::this_thread::get_id())
        std::cout << "This is the main thread.\n";
    else
        std::cout << "This is not the main thread.\n";
}

void pause_thread(int n) {
    std::this_thread::sleep_for(std::chrono::seconds(n));
    std::cout << "pause of " << n << " seconds ended\n";
}

int main() {
    std::thread t(hello);
    std::cout << t.hardware_concurrency() << std::endl;//可以并发执行多少个(不准确)
    std::cout << "native_handle " << t.native_handle() << std::endl;//可以并发执行多少个(不准确)
    t.join();
    std::thread a(hello);
    a.detach();
    std::thread threads[5];                         // 默认构造线程

    std::cout << "Spawning 5 threads...\n";
    for (int i = 0; i < 5; ++i)
        threads[i] = std::thread(pause_thread, i + 1);   // move-assign threads
    std::cout << "Done spawning threads. Now waiting for them to join:\n";
    for (auto &thread : threads)
        thread.join();
    std::cout << "All threads joined!\n";
}

Linux系统程序和应用程序的存放目录以及各目录之间的区别

系统的path环境变量中，至少应包含这些路径
开发软件应根据软件用途，将其存放在合理的目录中 

所有用户皆可使用的系统程序存放在/bin

头文件

函数库

函数库是以重复利用为目的，经过编译的函数集合。
标准的系统函数库都放在/lib和/usr/lib两个目录下，c语言编译器在链接时需要这些函数库。 

函数库名字总以lib开头，后面紧跟说明用途的字符串，然后跟一个'.',在后面部分指明函数库的类型：  

.a 静态链接函数库  
.so和.sa 共享函数库(动态链接函数库)  

无论静态库或是动态库，都由.o文件创建而来。过程是这样的：  

1. 编写函数库的源程序（.h文件和.c文件）
//hello.h  
#ifndef HELLO_H
#define HELLO_H
void hello(const char *name);
#endif  

//hello.c  
#include <stdio.h>
void hello(const char *name)
{
printf("Hello %s!\n", name);
}

2. 通过gcc先编译源代码得到.o 文件  
gcc -c hello.c  

3.1 由.o文件创建静态库   

ar -crv libmyhello.a hello.o  
静态库文件名的命名规范是以 lib 为前缀，紧接着跟静态库名，扩展名为.a

3.2 程序使用静态库  
第一步：在使用到这些公用函数的源程序中包含这些公用函数的原型声明  
第二步：用gcc命令生成目标文件时指明静态库名，gcc将会从静态库中将公用函数连接到目标文件中（用 gcc 命令生成目标文件时指明静态库名，gcc 将会从静态库中将公用函数连接到目标文件中）  

方式1: gcc main.c libmyhello.a -o hello
方式2: gcc -o hello main.c -L. -lmyhello  

//main.c  
#include "hello.h"
int main()
{
  hello("everyone");
  return 0;
}

第三步：静态函数库中的函数已经连接到目标文件里去了。  
直接运行 ./hello

4.1 由.o文件创建动态库 

gcc -shared -fPIC -o libmyhello.so hello.o  
生成libmyhello.so动态库  

4.2 程序使用动态库  
和使用静态库完全一样，也是在使用到这些公用函数的源程序中包含这些公用函数的原型声明，然后在用 gcc 命令生成目标文件时指明动态库名进行编译。  

gcc -o hello main.c -L. -lmyhello
稍有区别的是此时./hello无法运行  

原来是找不到动态库文件 libmyhello.so。程序在运行时，会在/usr/lib 或 /lib 等目录中查找需要的动态库文件。若找到，则载入动态库，否则将提示类似上述错误而终止程序运行。我们将文件 libmyhello.so 复制到目录 /lib 中，再试试，OK，也进一步说明了动态库在程序运行时是需要的

静态库在程序编译时会被连接到目标代码中，程序运行时将不再需要该静态库。动态库在程序编译时并不会被连接到目标代码中，而是在程序运行是才被载入，因此在程序运行时还需要动态库存在。

shell简介

Shell编程是linux下常用的扩展用户交互能力的方式，通过简单编程将常见命令扩展成灵活的新功能。
Shell有点类似window下的批处理，但又比批处理强大，有很多C语言和其他语言的特征，又没有其他语言那么复杂。

shell变量

linxu系统有3种变量:环境变量、内部变量、用户变量 

环境变量：  
如PATH,是系统自动为用户提供，用户可以对其中一些变量进行修改。  

内部变量：  
系统提供，用户可以使用，但不能修改它  

用户变量：
不像C(C++)中的变量，shell中的用户变量不需要声明，可以是数字型也可是字符串型
