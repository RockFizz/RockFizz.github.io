// *.pro:  
INCLUDEPATH += F:/Download/VTK/VTK/include/vtk-8.2
LIBS += F:/Download/VTK/VTK/lib/vtk*.lib
// *.h:  
#include <vtkAutoInit.h>
VTK_MODULE_INIT(vtkRenderingOpenGL2) // VTK was built with vtkRenderingOpenGL2
VTK_MODULE_INIT(vtkInteractionStyle)
VTK_MODULE_INIT(vtkRenderingFreeType)

#include <QVTKOpenGLNativeWidget.h>
#include <vtkSmartPointer.h>
#include <vtkRenderer.h>
#include <vtkRenderWindow.h>
#include <vtkSphereSource.h>
#include <vtkPolyDataMapper.h>
#include <vtkActor.h>
#include <vtkProperty.h>
#include <vtkRenderWindowInteractor.h>

// *.cpp:  
MainWindow::MainWindow(QWidget *parent) :
    QMainWindow(parent),
    ui(new Ui::MainWindow)
{
    ui->setupUi(this);

//    ui->widget1=new QVTKOpenGLNativeWidget(this);
    //创建渲染器和窗口
    vtkSmartPointer<vtkRenderer> renderer = vtkSmartPointer<vtkRenderer>::New();
    ui->widget1->GetRenderWindow()->AddRenderer(renderer);
    renderer->SetBackground(0.1, 0.1, 0.1);
    // 创建球体的源
    vtkSmartPointer<vtkSphereSource> sphereSource = vtkSmartPointer<vtkSphereSource>::New();
    sphereSource->SetRadius(1.0);
    // 创建球体的PolyDataMapper和Actor
    vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New();
    mapper->SetInputConnection(sphereSource->GetOutputPort());

    vtkSmartPointer<vtkActor> actor = vtkSmartPointer<vtkActor>::New();
    actor->SetMapper(mapper);
    actor->GetProperty()->SetColor(1.0, 0.0, 0.0); // 设置为红色
    // 将Actor加入渲染器
    renderer->AddActor(actor);
    renderer->ResetCamera();

    ui->widget1->show();
    setWindowTitle(tr("Hello World"));
}

// *.pro  
INCLUDEPATH += F:/Download/VTK/VTK/include/vtk-8.2

// *.h

// *.cpp  
MainWindow::MainWindow(QWidget *parent) :
    QMainWindow(parent),
    ui(new Ui::MainWindow)
{
    ui->setupUi(this);

    //ui->widget1=new QVTKOpenGLNativeWidget(this);
    //创建渲染器和窗口
    vtkSmartPointer<vtkRenderer> renderer = vtkSmartPointer<vtkRenderer>::New();
    ui->widget1->GetRenderWindow()->AddRenderer(renderer);
    renderer->SetBackground(0.1, 0.1, 0.1);
    // 创建立方体源
    vtkSmartPointer<vtkCubeSource> cubeSource = vtkSmartPointer<vtkCubeSource>::New();
    cubeSource->SetCenter(0, 0, 0);
    cubeSource->SetXLength(1);
    cubeSource->SetYLength(1);
    cubeSource->SetZLength(1);
    // 创建纹理读取器
    vtkSmartPointer<vtkJPEGReader> textureReader = vtkSmartPointer<vtkJPEGReader>::New();
    textureReader->SetFileName("D:/bird.jpg");
    textureReader->Update();
    // 创建纹理
    vtkSmartPointer<vtkTexture> texture = vtkSmartPointer<vtkTexture>::New();
    texture->SetInputConnection(textureReader->GetOutputPort());
    // 创建立方体的 PolyDataMapper 和 Actor
    vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New();
    mapper->SetInputConnection(cubeSource->GetOutputPort());

    vtkSmartPointer<vtkActor> actor = vtkSmartPointer<vtkActor>::New();
    actor->SetMapper(mapper);
    actor->SetTexture(texture);
    // actor->GetProperty()->SetColor(0, 0, 1); // 设置颜色为蓝色
    // 将Actor加入渲染器
    renderer->AddActor(actor);
    renderer->ResetCamera();

    ui->widget1->show();
    setWindowTitle(tr("Hello World"));
}

Qt_VTK使用合集

智能指针

msvc2015编译器默认支持c++11标准，std::shared_ptr是C++标准库中的智能指针，用于管理动态分配的对象。它提供了自动化的内存管理和资源释放，可以避免内存泄漏和野指针的问题 

#include <iostream>
#include <memory>
class MyClass {
public:
    MyClass() {
        std::cout << "MyClass constructor" << std::endl;
    }
    ~MyClass() {
        std::cout << "MyClass destructor" << std::endl;
    }
    void sayHello() {
        std::cout << "Helloworld" << std::endl;
    }
};

int main(int argc, char *argv[])
{
    // 创建一个 shared_ptr，指向一个动态分配的 MyClass 对象
    std::shared_ptr<MyClass> ptr(new MyClass());
    // 使用箭头运算符访问 MyClass 的成员函数
    ptr->sayHello();
    // 可以像普通指针一样使用 * 运算符解引用
    (*ptr).sayHello();
    // 共享所有权，拷贝构造函数会使引用计数增加
    std::shared_ptr<MyClass> ptr2 = ptr;
    // 引用计数为2，两个 shared_ptr 共享同一个对象
    // 重置指针，引用计数减少
    ptr.reset();
    // 引用计数为1，ptr 指向了空指针，ptr2 仍然指向原来的对象
    return 0;
}

protected:
    bool eventFilter(QObject *watched, QEvent *event)override{
        if (watched == this || watched->isWidgetType()) {
            if (event->type() == QEvent::MouseMove) {
                QMouseEvent *mouseEvent = static_cast<QMouseEvent*>(event);
                // QPoint pos = mouseEvent->globalPos();
                QPoint pos =mapFromGlobal(mouseEvent->globalPos());
                QString posText = QString("X: %1, Y: %2").arg(pos.x()).arg(pos.y());
                if(!m_statusBar) {
                    m_statusBar = new QStatusBar(this);
                    setStatusBar(m_statusBar);
                }
                m_statusBar->showMessage(posText);
            }
        }
        return QMainWindow::eventFilter(watched, event);

VTK7.1编译

下载VTK7.1.1源文件, https://vtk.org/files/release/7.1/VTK-7.1.1.zip 解压到F:\Download\VTK\VTK-7.1.1
下载VTK扩展数据, https://vtk.org/files/release/7.1/VTKData-7.1.1.zip解压到F:\Download\VTK\VTK-8.2.0目录下，会出现一个.ExternalData的文件夹
下载CMAKE,下载cmake-3.8.0-win64-x64.zip

启动CMake程序
指定source code路径为VTK解压路径（此处为F:\Download\VTK\VTK-7.1.1）  
build the binaries路径为cmake生成文件路径（此处为F:\Download\VTK\VTK-7.1.1\build 需要新建build文件夹）  
点击 configuration 按钮，弹窗选择编译器Visual Studio 14 2015 Win64  

BUILD：BUILD_SHARED_LIBS ->指示编译共享库还是静态库，默认是编译选上的状态，我们要编译共享库，所以保持默认状态；

vtk7.1显示一个红色球体

// *.pro:  
INCLUDEPATH += F:/Download/VTK/VTK/include/vtk-7.1
LIBS += F:/Download/VTK/VTK/lib/vtk*.lib
// *.h:  
#include <vtkAutoInit.h>
VTK_MODULE_INIT(vtkRenderingOpenGL2) // VTK was built with vtkRenderingOpenGL2
VTK_MODULE_INIT(vtkInteractionStyle)
VTK_MODULE_INIT(vtkRenderingFreeType)

#include <QVTKWidget.h>
#include <vtkSmartPointer.h>
#include <vtkRenderer.h>
#include <vtkRenderWindow.h>
#include <vtkSphereSource.h>
#include <vtkPolyDataMapper.h>
#include <vtkActor.h>
#include <vtkProperty.h>
#include <vtkRenderWindowInteractor.h>

// *.cpp:  
MainWindow::MainWindow(QWidget *parent) :
    QMainWindow(parent),
    ui(new Ui::MainWindow)
{
    ui->setupUi(this);

//    ui->widget1=new QVTKOpenGLNativeWidget(this);
    //创建渲染器和窗口
    vtkSmartPointer<vtkRenderer> renderer = vtkSmartPointer<vtkRenderer>::New();
    ui->widget1->GetRenderWindow()->AddRenderer(renderer);
    renderer->SetBackground(0.1, 0.1, 0.1);
    // 创建球体的源
    vtkSmartPointer<vtkSphereSource> sphereSource = vtkSmartPointer<vtkSphereSource>::New();
    sphereSource->SetRadius(1.0);
    // 创建球体的PolyDataMapper和Actor
    vtkSmartPointer<vtkPolyDataMapper> mapper = vtkSmartPointer<vtkPolyDataMapper>::New();
    mapper->SetInputConnection(sphereSource->GetOutputPort());

    vtkSmartPointer<vtkActor> actor = vtkSmartPointer<vtkActor>::New();
    actor->SetMapper(mapper);
    actor->GetProperty()->SetColor(1.0, 0.0, 0.0); // 设置为红色
    // 将Actor加入渲染器
    renderer->AddActor(actor);
    renderer->ResetCamera();

    ui->widget1->show();
    setWindowTitle(tr("Hello World"));
}
