加载配置xml

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

using System.IO;
using System.Xml;
public class CalcXml : MonoBehaviour {


    void Start () {
        xmlPath = Application.dataPath + "/StreamingAssets/config.xml";
        CreateXml();

    }
    string xmlPath="";


    void Update () {

    }
    void OnGUI() {
        if (GUI.Button(new Rect(Screen.width - 110, 50, 100, 30), "xml播放视频"))
        {
            ReadVideo();
        }
        if (GUI.Button(new Rect(Screen.width - 110, 150, 100, 30), "xml播放音频"))
        {
            ReadAudio();
        }
        if (GUI.Button(new Rect(Screen.width - 110, 200, 100, 30), "xml播放图片"))
        {
            ReadSprite();
        }
        return;
        if (GUI.Button(new Rect(Screen.width - 100, 10, 90, 30), "改动xml")) {
            UpdateXml("fdfag");
        }
    }
    void CreateXml() {

        if (!File.Exists(xmlPath)) {
            XmlDocument xmlDoc = new XmlDocument();
            XmlElement data = xmlDoc.CreateElement("data");
            xmlDoc.AppendChild(data);
            XmlElement color = xmlDoc.CreateElement("color");
            data.AppendChild(color);
            xmlDoc.Save(xmlPath);
            Debug.Log("创建XML完成");
        }
    }
    void UpdateXml(string color) {
        if (File.Exists(xmlPath))
        {
            XmlDocument xmlDoc = new XmlDocument();
            xmlDoc.Load(xmlPath);
            XmlNode colorNode = xmlDoc.SelectSingleNode("data/color");
            colorNode.InnerText = color.ToString();
            xmlDoc.Save(xmlPath);
            Debug.Log("改动成功：" + color);
        }
    }
    void ReadVideo() {
        if (File.Exists(xmlPath)) {
            XmlDocument xmlDoc = new XmlDocument();
            xmlDoc.Load(xmlPath);
            XmlNodeList node = xmlDoc.SelectSingleNode("data/videos").ChildNodes;
            foreach (XmlElement ele in node) {
                if (ele.Name == "video") {
                    if (ele.GetAttribute("id") == "1") {
                        string _fold = ele.GetAttribute("fold");
                        string _name = ele.GetAttribute("name");
                        GameObject.FindObjectOfType<MoviePlayByRawImage>().PlayVideoByName(_fold,_name);
                    }
                }
            }
        }
    }


    void ReadAudio()
    {
        if (File.Exists(xmlPath))
        {
            XmlDocument xmlDoc = new XmlDocument();
            xmlDoc.Load(xmlPath);
            XmlNodeList node = xmlDoc.SelectSingleNode("data/audios").ChildNodes;
            foreach (XmlElement ele in node)
            {
                if (ele.Name == "audio")
                {
                    if (ele.GetAttribute("id") == "j")
                    {
                        string _fold = ele.GetAttribute("fold");
                        string _name = ele.GetAttribute("name");
                        GameObject.FindObjectOfType<AudioPlayFromStreamingAsset>().PlayAudioByName(_fold, _name);
                    }
                }
            }
        }
    }

    void ReadSprite()
    {
        if (File.Exists(xmlPath))
        {
            XmlDocument xmlDoc = new XmlDocument();
            xmlDoc.Load(xmlPath);
            XmlNodeList node = xmlDoc.SelectSingleNode("data/sprites").ChildNodes;
            foreach (XmlElement ele in node)
            {
                if (ele.Name == "sprite")
                {
                    if (ele.GetAttribute("id") == "zz")
                    {
                        string _fold = ele.GetAttribute("fold");
                        string _name = ele.GetAttribute("name");
                        GameObject.FindObjectOfType<SpriteFromStreamingAsset>().LoadSpriteByName(_fold, _name);
                    }
                }
            }
        }
    }
}

    {

    private IEnumerator playVideo(int _videoIndex = 0)
    {
        nowVideoIndex = _videoIndex;
        nowVideoName = localVideoClips[_videoIndex].name;

        Handlev_videoPlayCompleted = null;
        Handlev_videoPlayStart = null;
        Handlev_videoPlayCompleted = LocalVideoComplete;
        Handlev_videoPlayStart = LocalVideoStart;



        _videoPlayer.source = VideoSource.VideoClip;
        _videoPlayer.clip = localVideoClips[_videoIndex];
        _videoPlayer.audioOutputMode = VideoAudioOutputMode.AudioSource;
        _videoPlayer.EnableAudioTrack(0, true);
        _videoPlayer.SetTargetAudioSource(0, _audioSource);
        _videoPlayer.Prepare();

        _rawImage.color = new Color(0, 0, 0);
        while (!_videoPlayer.isPrepared)
        {
            yield return new WaitForSeconds(0.05f);
        }
        _rawImage.color = new Color(1, 1, 1);

       // Debug.Log("视频 宽：" + _videoPlayer.clip.width + "  高：" + _videoPlayer.clip.height);

        string timeVideoAll = Mathf.FloorToInt((float)(_videoPlayer.frameCount / _videoPlayer.frameRate)).ToString() + "  " + nowVideoName + "  本地视频编号:" + _videoIndex.ToString();
        Debug.Log("视频总时间：" + timeVideoAll);
        if (Handlev_videoPlayStart != null)
        {
            Handlev_videoPlayStart(_videoIndex);
        }


        _rawImage.texture = _videoPlayer.texture;
        _videoPlayer.Play();
        _audioSource.Play();
        while (_videoPlayer.isPlaying)
        {
            Debug.Log("视频当前播放:" + Mathf.FloorToInt((float)_videoPlayer.time) + "/" + timeVideoAll);
            yield return null;
        }
        Debug.Log("视频播放结束");

        if (Handlev_videoPlayCompleted != null)
        {
            Handlev_videoPlayCompleted(_videoIndex);
        }

    }
    private IEnumerator playVideoURL(string _url)
    {

        _videoPlayer.source = VideoSource.Url;
        _videoPlayer.url = _url;
        _videoPlayer.audioOutputMode = VideoAudioOutputMode.AudioSource;
        _videoPlayer.EnableAudioTrack(0, true);
        _videoPlayer.SetTargetAudioSource(0, _audioSource);
        _videoPlayer.Prepare();
        _rawImage.color = new Color(0, 0, 0);
        while (!_videoPlayer.isPrepared)
        {
            yield return new WaitForSeconds(0.05f);
        }
        _rawImage.color = new Color(1, 1, 1);



        nowVideoName = _videoPlayer.source.ToString();
        string timeVideoAll = Mathf.FloorToInt((float)(_videoPlayer.frameCount / _videoPlayer.frameRate)).ToString();
        Debug.Log("视频总时间：" + timeVideoAll);

        _rawImage.texture = _videoPlayer.texture;
        _videoPlayer.Play();
        _audioSource.Play();
        while (_videoPlayer.isPlaying)
        {
            Debug.Log("视频当前播放:" + Mathf.FloorToInt((float)_videoPlayer.time) + "/" + timeVideoAll);
            yield return null;
        }
        Debug.Log("视频播放结束");


    }

    //PC端读取StreamingAssets文件夹内音频，支持ogg格式。
    private AudioSource _audioSource; 
    void Start () {
        _audioSource = this.GetComponent<AudioSource>();

    }


    void Update () {

    }

    IEnumerator readAudioFromStreamingAsset(string _fold,string _name) {
        WWW _www = new WWW("file://" + Application.dataPath + "/StreamingAssets/"+_fold+"/" + _name+".ogg");
        yield return _www;
        if (_www.error != null) {
            Debug.Log(_www.error);
        }
        AudioClip ac = _www.GetAudioClip();
        _audioSource.clip = ac;
        _audioSource.Play();
    }
    public void PlayAudioByName(string fold, string name)
    {         
        StartCoroutine(readAudioFromStreamingAsset(fold, name));
    }
}

        for (int i = 0; i < Length; i ++)
        {
            newRandom.Append(constant[rd.Next(62)]); //rd.Next(62)返回小于62的非负随机数,Append将Length次随机的码进行拼接

    }
    public Text currentTimeText;
    private int hour;
    private int minute;
    private int second;
    private int year;
    private int month;
    private int day;
    void Update () {
        if (currentTimeText != null) {
            currentTimeText.text = formatDatetime();
        }
    }
    string formatDatetime() {
        hour = System.DateTime.Now.Hour;
        minute = System.DateTime.Now.Minute;
        second = System.DateTime.Now.Second;
        year = System.DateTime.Now.Year;
        month = System.DateTime.Now.Month;
        day = System.DateTime.Now.Day;
        return string.Format("{3:D4}/{4:D2}/{5:D2}"+"  "+"{0:D2}:{1:D2}:{2:D2} " , hour, minute, second, year, month, day);
    }
}

飞行视角相机

using UnityEngine;
using System.Collections;


public class FlyCam : MonoBehaviour
{
    public float sensitivityX = 5F;
    public float sensitivityY = 5F;

    public float minimumX = -360F;
    public float maximumX = 360F;

    public float minimumY = -90F;
    public float maximumY = 90F;

    public float smoothSpeed = 20F;

    float rotationX = 0F;
    float smoothRotationX = 0F;
    float rotationY = 0F;
    float smoothRotationY = 0F;
    Vector3 vMousePos;

    bool bActive = false;

    void Start()
    {
        vMousePos = Input.mousePosition;
        if (Screen.fullScreen)
            Cursor.visible = false;
    }

    void Update()
    {
        if (Input.GetMouseButton(1))
        {
            rotationX += Input.GetAxis("Mouse X") * sensitivityX;
            rotationY += Input.GetAxis("Mouse Y") * sensitivityY;
            rotationY = Mathf.Clamp (rotationY, minimumY, maximumY);
            Cursor.visible = false;
            Screen.lockCursor = true;
            bActive = true;
        }
        else if (bActive || vMousePos != Input.mousePosition)
        {
            Cursor.visible = true;
            Screen.lockCursor = false;
        }

        vMousePos = Input.mousePosition;

        // smooth mouse look
        smoothRotationX += (rotationX - smoothRotationX) * smoothSpeed * Time.smoothDeltaTime;
        smoothRotationY += (rotationY - smoothRotationY) * smoothSpeed * Time.smoothDeltaTime;

        // transform camera to new direction

IEnumerator load(string url)//协程  
{  
    WWW www = new WWW(url);  
    yield return www;  
    if (www.error == null || www.error.Length <= 0)  
    {  
        string name = url.Substring(url.LastIndexOf('/') + 1);  
        //Application.dataPath + "/UI/" + name=加载路径  
        File.WriteAllBytes(Application.dataPath + "/UI/" + name, www.bytes);  
    }  
    else  
    {  
        print("加载出错"+www.error);  
    }  
    NextLoad();  
}
