每隔5秒，弹一个会自动关闭的非阻塞式的信息弹框

# coding:utf-8
import Tkinter as tk # Python2.7

def close_message_box():
    message_box.destroy()

def popup_message():
    global message_box
    message_box = tk.Toplevel()
    message_box.title("提醒")
    message_label = tk.Label(message_box, text="该休息一下,走动一下")
    message_label.pack()
    message_box.after(5000, close_message_box) # 5000毫秒（即5秒）后自动关闭弹窗
    root.after(8000,popup_message) # 8000毫秒（即8秒）后再次弹窗

root = tk.Tk()
# root.withdraw()  # 隐藏主窗口
root.after(8000,popup_message)
root.mainloop()

通过一系列点拟合绘制曲线

import numpy as np
import matplotlib.pyplot as plt

# 准备数据
x = np.array([1, 2, 3, 4, 5])
y = np.array([2.3, 4.9, 8.3, 12.5, 17.5])

# 拟合数据
# 使用 np.polyfit() 函数拟合数据。
# 这个函数可以根据提供的 x 和 y 值拟合出一个 n 次多项式，
# 返回一个包含多项式系数的数组。
# 在这里我们拟合一条二次函数，即 n=2。

coef = np.polyfit(x, y, 2)  # 拟合二次函数
poly = np.poly1d(coef) # 创建一个多项式对象，并传入拟合得到的系数

# 计算新数据点
new_x = np.linspace(x[0], x[-1], 50)  # 生成一个包含 50 个元素的等间距数组，范围从 x 的最小值到最大值
new_y = poly(new_x)

# 绘制图像
plt.scatter(x, y, color='blue')  # 绘制数据点
plt.plot(new_x, new_y, color='red')  # 绘制拟合曲线
plt.show()
