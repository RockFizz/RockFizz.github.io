(Unity3d)环绕观察的观察点平滑切换

使用如下脚本，挂到Camera上，Target为被观察点的初始化位置(该Transform可见，不可见都可以)，运行后修改TargetTo的位置，实现环绕观察的观察点平滑切换。 

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class See3DModelOneToOne : MonoBehaviour {

    public Transform target;
    public Transform targetTO;
    public Vector3 targetOffset;
    public float distance = 5.0f;
    public float maxDistance = 20;//缩放
    public float minDistance = .6f;//缩放
    public float xSpeed = 200.0f;//速度
    public float ySpeed = 200.0f;
    public int yMinLimit = -80;//限定角度
    public int yMaxLimit = 80;
    public int zoomRate = 40;
    public float zoomDampening = 5.0f;
    public float xDeg = 0.0f;//自身的角度记录
    public float yDeg = 0.0f;//自身的角度
    public float currentDistance;//缩放记录
    public float desiredDistance;//缩放
    private Quaternion currentRotation;
    private Quaternion desiredRotation;
    public Quaternion rotation;
    private Vector3 position;

    public float X;//初始化角度X轴
    public float Y;//初始化角度Y轴
    /// <summary>
    /// 初始化距离
    /// </summary>
    public float CameDistance;
    void Start()
    {
        Init();
        desiredDistance = CameDistance;
        desiredDistance = 6.8f;
        yDeg = X;
        xDeg = Y;
    }
    void OnEnable()
    {
        Init();
        desiredDistance = CameDistance;
        desiredDistance = 6.8f;
        yDeg = X;
        xDeg = Y;

    }

    public void Init()
    {

        if (!target)
        {
            GameObject go = new GameObject("Cam Target");
            go.transform.position = transform.position + (transform.forward * distance);
            target = go.transform;
        }

        distance = Vector3.Distance(transform.position, target.position);
        distance = 6.8f;
        currentDistance = distance;
        desiredDistance = distance;


        position = transform.position;
        rotation = transform.rotation;
        currentRotation = transform.rotation;
        desiredRotation = transform.rotation;

        xDeg = Vector3.Angle(Vector3.right, transform.right);
        yDeg = Vector3.Angle(Vector3.up, transform.up);
    }


    void LateUpdate()
    {

        if (Input.GetMouseButton(0))
        {
            xDeg += Input.GetAxis("Mouse X") * xSpeed * 0.02f;
            yDeg -= Input.GetAxis("Mouse Y") * ySpeed * 0.02f;
        }
        yDeg = ClampAngle(yDeg, yMinLimit, yMaxLimit);
        // 设置相机旋转

        desiredRotation = Quaternion.Euler(yDeg, xDeg, 0);
        currentRotation = transform.rotation;


        rotation = Quaternion.Lerp(currentRotation, desiredRotation, Time.deltaTime * zoomDampening);
        transform.rotation = rotation;



        // 影响scrollwheel变焦距离
        desiredDistance -= Input.GetAxis("Mouse ScrollWheel") * Time.deltaTime * zoomRate * Mathf.Abs(desiredDistance);
        //变焦最小/最大
        desiredDistance = Mathf.Clamp(desiredDistance, minDistance, maxDistance);
        //平滑变焦
        currentDistance = Mathf.Lerp(currentDistance, desiredDistance, Time.deltaTime * zoomDampening);


        target.position = Vector3.Lerp(target.position, targetTO.position, Time.deltaTime * 5);


        position = target.position - (rotation * Vector3.forward * currentDistance + targetOffset);
        transform.position = position;
    }

    private static float ClampAngle(float angle, float min, float max)
    {
        if (angle < -360)
            angle += 360;
        if (angle > 360)
            angle -= 360;
        return Mathf.Clamp(angle, min, max);
    }

}

Unity3d鼠标对3D物体的点击和悬浮

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class ClickedInfo : MonoBehaviour
{
    public Camera _mainCamera;

    void Update()
    {
        RaycastHit hit;
        if (Physics.Raycast(_mainCamera.ScreenPointToRay(Input.mousePosition), out hit))
        {
            bool cube = false;
            bool sphere = false;
            bool cylinder = false;
            //鼠标悬浮的操作（鼠标点击场景任何地方后）
            if (hit.collider.gameObject.name == "Cube1")
            {
                Debug.Log("Mouse on the Cube");
                cube = true;
            }
            else if (hit.collider.gameObject.name == "Sphere")
            {
                Debug.Log("Mouse on the Sphere");
                sphere = true;
            }
            else if (hit.collider.gameObject.name == "Cylinder")
            {
                Debug.Log("Mouse on the Sphere");
                cylinder = true;
            }
            else
            {
                Debug.Log("Mouse on Nothing");
            }
            //鼠标点击的操作
            if (Input.GetMouseButtonDown(0))
            {
                if (cube)
                {
                    Debug.Log("CUBE");
                }
                else if (sphere)
                {
                    Debug.Log("SPHERE");
                }
                else if (cylinder)
                {
                    Debug.Log("CYLINDER");
                }
                else
                {
                    Debug.Log("NOTHING!");
                }
            }
        }
    }

}

Unity3d的GUI跟随鼠标

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class WithMouse : MonoBehaviour
{

    public Texture2D Mid;
    public Texture2D mouse;
    public Vector2 mousePs;

    private float angle;
    private Vector2 FixedPoint1PosV2;
    void Start()
    {
        FixedPoint1PosV2 = new Vector2(250,250);
    }


    void Update()
    {
        //获取屏幕上鼠标位置
        mousePs = Input.mousePosition;
        //获得实时鼠标位置相对于某固定点的角度
        angle = 180 - Mathf.Atan2(mousePs.x - FixedPoint1PosV2.x, Screen.height - mousePs.y - FixedPoint1PosV2.y) * Mathf.Rad2Deg;//计算选择角度
    }
    private void OnGUI()
    {
        //GUI上旋转，使之对齐鼠标点
        GUIUtility.RotateAroundPivot(angle, new Vector2(250, 250));
        GUI.DrawTexture(new Rect(200, 200, 100, 100), Mid);
        GUIUtility.RotateAroundPivot(-angle, new Vector2(250, 250));

        //GUI上，跟随鼠标点显示
        GUI.DrawTexture(new Rect(mousePs.x, Screen.height - mousePs.y, 237, 60), mouse);
    }
}

Unity3d点击物体弹窗(内置GUI)

OnMouseEnter、OnMouseDown、OnMouseExit依赖MonoBehaviour(原理是射线)，鼠标触发这些时间需要将本脚本绑定到带碰撞的目标物体本身上，如果有父物体，父物体添加了刚体或者碰撞体后同样会造成检测不到。

using System.Collections;

using UnityEngine;

public class ClickObjPopupWin : MonoBehaviour
{
    public GameObject target;
    public bool WindowSwitch = false;
    private Rect windowRect =new Rect(200, 80, 240, 100);
    private Vector2 mousePs;
    private void OnGUI()
    {
        if (WindowSwitch) {            
            windowRect = GUI.Window(0, windowRect, WindowContain, "弹出窗口");
        }

    }
    void WindowContain(int windowID) {
        if (GUI.Button(new Rect(70, 40, 100, 20), "关闭窗口"))
        {
            WindowSwitch = false;
        }
    }
    /*    
OnMouseEnter、OnMouseDown、OnMouseExit依赖MonoBehaviour(原理是射线)，鼠标触发这些时间需要将本脚本绑定到带碰撞的目标物体本身上，如果有父物体，父物体添加了刚体或者碰撞体后同样会造成检测不到。
    */

    }

        //获取屏幕上鼠标位置
        mousePs = Input.mousePosition;
    }

}

新版Unity3d基于jslib和网页js方法互相互调

新版Unity弃用了Application.ExternalCall()，加了个jslib当中介。Unity必须通过jslib来调用网页的js方法。然而js还是通过.SendMessage()调用Unity方法！

c#传字符串给js时，需要调用unity3d自带函数Pointer_stringify将字符串转化一下，才能被js识别。传递int、float不需要转化直接传

  1. 添加jslib文件。(必须放在Unity项目的Plugins目录内) 
//Plugins下Js_Invoke.jslib:  

mergeInto(LibraryManager.library, {
    HelloFun: function () {  //给unity3d识别的函数名
        TestSendFunInJS();    //JS里真实的函数名
    },
    CallWin:function(_x,_y,_name){
        CallWinInJS(_x,_y,Pointer_stringify(_name));
    }
});
  1. 创建c#脚本

注意添加”using System.Runtime.InteropServices;” 

using UnityEngine;

Unity3D开发之webgl键盘事件监听和其他js输入冲突

做一个大的集控平台的时候，很多时候会把unity导出的webgl3D程序嵌到大平台页面中，属于大平台的一个子功能。在打开3d程序之后再切换到其他js页面的时候，我们会发现其他页面的输入键盘事件不好使了。比如一个input无法输入，也无法删除input里的文字。即input事件被劫听了。
情况下unity的webgl会监听所有的键盘事件，无论你是否聚焦这个canvas，只要你打开程序，我们就监听。目的是为了让玩家更加专注游戏而不需要去点击聚焦程序。但是如果你还有其他的html页面，并且其他页面也需要监听这个事件，那就需要改变这个默认设置。使用WebGLInput.captureAllKeyboardInput这个属性。所以我们在程序中Awake函数中加入设置这个属性的代码。 

private void Start()
{
#if !UNITY_EDITOR && UNITY_WEBGL 
     UnityEngine.WebGLInput.captureAllKeyboardInput = false;
#endif
}

Unity3d选择、移动3d物体

将以下脚本附加到目标物体(带碰撞)即可实现：点击选择物体，并拖拽物体移动。 

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class PickAndMove : MonoBehaviour
{

    public Renderer rend;
    public Vector3 targetPos;
    public Vector3 lastPos;

    private bool isdrag = false;
    void Start()
    {
        rend = GetComponent<Renderer>();
    }

    private void OnMouseEnter()
    {
        if (!isdrag) {
            rend.material.color = Color.blue;
        }
    }
    private void OnMouseExit()
    {
        if (!isdrag) { 
            rend.material.color = Color.red;
        }
    }
    private void OnMouseDown()
    {
        if (!isdrag)
        {
            rend.material.color = Color.yellow;
            lastPos = Camera.main.ScreenToViewportPoint(Input.mousePosition) - Camera.main.WorldToViewportPoint(transform.position);

        }
    }
    private void OnMouseUp()
    {
        rend.material.color = Color.blue;
        isdrag = false;
    }
    private void OnMouseDrag()
    {
        targetPos = Camera.main.ViewportToWorldPoint(Camera.main.ScreenToViewportPoint(Input.mousePosition) - lastPos);
        transform.position = new Vector3(targetPos.x, targetPos.y, targetPos.z);
        isdrag = true;
    }

    void Update()
    {

    }
}
